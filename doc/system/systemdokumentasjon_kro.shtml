CREATE TABLE datafeil (
    stationid integer NOT NULL,
    obstime timestamp with time zone NOT NULL,
    original double precision NOT NULL,
    paramid integer NOT NULL,
    tbtime timestamp with time zone NOT NULL,
    typeid integer NOT NULL,
    sensor character(1) DEFAULT '0',
    "level" integer,
    corrected double precision,
    controlinfo character(16) DEFAULT '0000000000000000',
    useinfo character(16) DEFAULT '0000000000000000',
    cfailed text,
    tack timestamp with time zone DEFAULT ('now'::text)::timestamp(6) with time zone,
    UNIQUE (stationid, obstime, paramid, level, sensor, typeid)
);


CREATE TABLE feil (
    nr integer DEFAULT nextval('"feil_nr_seq"'::text) NOT NULL,
    stasjonsnr integer NOT NULL,
    feiltypenr integer NOT NULL,
    tiltaksnr integer,
    kvittering boolean DEFAULT false,
    signaturnr integer,
    lestid timestamp with time zone,
    CONSTRAINT feil_pkey PRIMARY KEY (nr),
    CONSTRAINT feil_lestid CHECK (NULL::boolean)
);


CREATE TABLE feiltype (
    nr integer DEFAULT nextval('"feiltype_nr_seq"'::text) NOT NULL,
    tekst character varying(100) NOT NULL,
    CONSTRAINT feiltype_tekst_key UNIQUE (tekst),
    CONSTRAINT feiltype_pkey PRIMARY KEY (nr)
);


CREATE TABLE infomelding (
    nr integer DEFAULT nextval('"infomelding_nr_seq"'::text) NOT NULL,
    stasjonsnr integer NOT NULL,
    tekst character varying(1000) NOT NULL,
    infostatus integer DEFAULT 2,
    tid timestamp with time zone DEFAULT ('now'::text)::timestamp(6) with time zone,
    tid_ok timestamp with time zone,
    signaturnr integer,
    signaturnr_ok integer,
    CONSTRAINT infomelding_pkey PRIMARY KEY (nr)
);


CREATE TABLE kvalobstest (
    nr integer DEFAULT nextval('"kvalobstest_nr_seq"'::text) NOT NULL,
    navn character varying(100) NOT NULL,
    antalltekst integer,
    tekst character varying(100) DEFAULT 'Ingen beskrivelse foreligger...' NOT NULL,
    CONSTRAINT kvalobstest_pkey PRIMARY KEY (nr)
);


CREATE TABLE kvalobstype (
    nr integer NOT NULL,
    feiltypenr integer,
    CONSTRAINT kvalobstype_pkey PRIMARY KEY (nr)
);


CREATE TABLE lokal (
    nr integer DEFAULT nextval('"lokal_nr_seq"'::text) NOT NULL,
    ip_adr inet NOT NULL,
    lest_stasjon character varying(100) NOT NULL,
    CONSTRAINT lokal_pkey PRIMARY KEY (nr)
);


CREATE TABLE lokal2 (
    nr integer DEFAULT nextval('"lokal2_nr_seq"'::text) NOT NULL,
    ip_adr inet NOT NULL,
    lest_stasjon character varying(100) NOT NULL,
    st_type character(1) NOT NULL,
    brukernavn character(50),
    CONSTRAINT lokal2_nr_key UNIQUE (nr)
);


CREATE TABLE melding (
    nr integer DEFAULT nextval('"melding_nr_seq"'::text) NOT NULL,
    tekst character varying(1000) NOT NULL,
    feilnr integer NOT NULL,
    tid timestamp with time zone DEFAULT ('now'::text)::timestamp(6) with time zone,
    signaturnr integer,
    paramnr integer NOT NULL,
    paramverdi numeric(10,5),
    testnr integer NOT NULL,
    testparam numeric[],
    CONSTRAINT melding_pkey PRIMARY KEY (nr)
);


CREATE TABLE parameter (
    nr integer NOT NULL,
    kode character(10) NOT NULL,
    definisjon character varying(100),
    kategori character varying(100),
    kategoritekst character varying(100),
    CONSTRAINT parameter_kode_key UNIQUE (kode),
    CONSTRAINT parameter_nr_key UNIQUE (nr),    
    CONSTRAINT parameter_pkey PRIMARY KEY (nr, kode)
);


CREATE TABLE signatur (
    nr integer DEFAULT nextval('"signatur_nr_seq"'::text) NOT NULL,
    navn character(100) NOT NULL,
    tilgang integer,
    kommentar character(100),
    ip_adr inet NOT NULL,
    mailadresse character(50),
    brukernavn character(50),
    CONSTRAINT signatur_pkey PRIMARY KEY (nr)
);


CREATE TABLE status (
    nr integer NOT NULL,
    farge_norsk_txt character varying(100),
    farge_eng_txt character varying(100),
    CONSTRAINT status_pkey PRIMARY KEY (nr)
);


CREATE TABLE stasjon (
    nr integer NOT NULL,
    navn character varying(100) NOT NULL,
    lat numeric(6,4),
    lon numeric(6,4),
    synopnr integer,
    a boolean DEFAULT false,
    f boolean DEFAULT false,
    h boolean DEFAULT false,
    i boolean DEFAULT false,
    n boolean DEFAULT false,
    t boolean DEFAULT false,
    v boolean DEFAULT false,
    region character varying(50),
    status integer DEFAULT 1,
    CONSTRAINT stasjon_pkey PRIMARY KEY (nr),
    CONSTRAINT stasjon_lat CHECK (((lat > -90.0000) AND (lat < 90.0000))),
    CONSTRAINT stasjon_lon CHECK (((lon > -90.0000) AND (lon < 90.0000))),
    CONSTRAINT stasjon_region CHECK ((((((((((region = 'Øst'::character varying) OR NULL::boolean) OR (region = 'Vest'::character varying)) OR (region = 'Nord'::character varying)) OR (region = 'Sør'::character varying)) OR (region = 'MidtNorge'::character varying)) OR (region = 'Maritim'::character varying)) OR (region = 'Test'::character varying)) OR (region = 'Arktis'::character varying))),
    CONSTRAINT "<unnamed>" FOREIGN KEY (status) REFERENCES status (nr) ON DELETE NO ACTION ON UPDATE NO ACTION NOT DEFERRABLE INITIALLY IMMEDIATE
);


CREATE TABLE stasjon_inne (
    stationid integer NOT NULL,

CREATE TABLE stasjon_nede (

CREATE TABLE tiltakstype (
    nr integer DEFAULT nextval('"tiltakstype_nr_seq"'::text) NOT NULL,
    tekst character varying(100) NOT NULL,
    CONSTRAINT tiltakstype_pkey PRIMARY KEY (nr)
);


CREATE TABLE tiltak (
    nr integer DEFAULT nextval('"tiltak_nr_seq"'::text) NOT NULL,
    tekst character varying(1000) NOT NULL,
    tiltaktypenr integer NOT NULL,
    feilnr integer NOT NULL,
    beskrevet boolean DEFAULT false,
    beskrivtid timestamp with time zone,
    signaturnr integer NOT NULL,
    ferdig boolean DEFAULT false,
    slutttid timestamp with time zone,
    sluttfarnr integer,
    CONSTRAINT tiltak_pkey PRIMARY KEY (nr),
    CONSTRAINT tiltak_tiltaktypenr FOREIGN KEY (tiltaktypenr) REFERENCES tiltakstype (nr) ON DELETE RESTRICT ON UPDATE NO ACTION NOT DEFERRABLE INITIALLY IMMEDIATE,
    CONSTRAINT tiltak_signaturnr FOREIGN KEY (signaturnr) REFERENCES signatur (nr) ON DELETE RESTRICT ON UPDATE NO ACTION NOT DEFERRABLE INITIALLY IMMEDIATE
);



DOKUMENTASJON AV TABELLENE.

For tabeller i bruk pr. idag:
Vi har de statiske tabellene feiltype, tiltakstype, status og signatur. Signatur vil dog endres når nye personer logger seg på.
De dynamiske tabellene er feil (inneholder alle feilsituasjoner), melding (alle meldingstekster generert), tiltak (alle beskrevne tiltak, boolsk verdi for bekreftet feil/feil rettet), stasjon (inneholder alle stasjoner, oppdateres mot st_info i klimadatabasen) og lokal2 (benyttes som midlertidig lagring av valgt stasjonsnavn til bruk i skript).


feil:
 KILDE: Oppdateres manuelt i KRO
HVEM bruker tabellen: Alle de andre tabellene i tiltaksdatabasen
BESKRIVELSE av tabellen: Dette er den sentrale tabellen i tiltaksdatabasen , som kobler feiltype, stasjonsnummer, melding, tiltak og signatur.
BESKRIVELSE av attributtene i tabellen:
nr: Nummeret på feilsituasjonen
stasjonsnr: Stasjonsnummeret
feiltypenr: Nummeret på feiltypen
tiltaksnr: Nummeret på eventuelt tiltak som er beskrevet
kvittering: Boolsk verdi ('t'/'f') for om feilsituasjonen er bekreftet eller ikke
signaturnr: Signaturnummer på vedkommende som har bekreftet feilsituasjonen
lestid: Tidspunkt for når feilsituasjonen er bekreftet (er tom hvis feilsituasjonen ikke er bekreftet)


feiltype:
 KILDE: Statisk tabell, kan endres manuelt i tiltaksdatabasen.
HVEM bruker tabellen: Feil
BESKRIVELSE av tabellen: Inneholder alle feiltypekategoriene med tilhørende nr.
BESKRIVELSE av attributtene i tabellen:
nr: nummer på feiltype
tekst: aktuell feiltype


infomelding:
KILDE: Oppdateres manuelt i KRO.
HVEM bruker tabellen: Perlskript for å vise registrert infomelding.
BESKRIVELSE av tabellen: Inneholder alle infomeldinger av forskjellig slag (info om fravær, praktiske opplysninger.)
BESKRIVELSE av attributtene i tabellen:
nr: Løpenummer i tabellen
stasjonsnr: Stasjonsnummeret infomeldingen hører til
tekst: Infomeldingens innhold
infostatus: Infostatus 2 (Default): betyr at infomeldingen vises i 'Stasjonsstatus-visning'. Status1 (kan settes av superbruker,: betyr at infomeldingen kun
vises i 'Historikk-visning'.
tid: Tid for infomeldingen
tid_ok: Ikke i bruk
signaturnr: Sigaturnummeret på den som står bak infomeldingen.
signaturnr_ok: ikke i bruk


lokal2:
KILDE: Tabell som fungerer som temporær lagring ved klikk på stasjon i KRO, slik at stasjonen "huskes". Denne plasseres da temporært i en ny rad.
HVEM bruker tabellen: Perlskriptet som viser resultatet av valget ditt.
BESKRIVELSE av tabellen: Brukes som mellomtabell for valgt stasjon,
dersom man trykker på "Registrering av ny melding", "Historikk" etc. så "huskes" stasjonen
man har valgt ved hjelp av denne tabellen (stasjonsvalget blir altså lagret midlertidig)
BESKRIVELSE av attributtene i tabellen:
nr: Løpenummer
ip_adr: IP-addressen til vedkommende som er inne på aktuell stasjon
lest_stasjon: Aktuell stasjon man har valgt
st_type: Type stasjon
brukernavn: Brukernavn


melding:
KILDE: Oppdateres manuelt i KRO
HVEM bruker tabellen: Perlskript for å vise registrert melding
BESKRIVELSE av tabellen: Denne tabellen inneholder alle meldinger som er registrert, med signatur, feilnummer og tidspunkt
BESKRIVELSE av attributtene i tabellen:
nr: løpenummeret på meldingen
tekst: meldingsteksten
feilnr: nummeret på feilsituasjonen som meldingen tilhører
tid: tid for innkommet melding
signaturnr: signaturnummeret til meldingskilden
paramnr:          Ikke i bruk
paramverdi:     Ikke i bruk
testnr:               Ikke i bruki 
testparam:        Ikke i bruk


signatur:
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Feil, Melding, Tiltak
BESKRIVELSE av tabellen: Signaturtabell. Inneholder alle registrerte brukernavn som har koblet seg på via LDAP
BESKRIVELSE av attributtene i tabellen:
nr: Signaturnummer
navn: Brukernavn
tilgang: Tall som angir rettigheter (1,2,3,4,5)
kommentar: Kommentarfelt
ip_adr: Brukerens ip-addresse


stasjon:
KILDE: Klimadatabasen, senere St_Info_Sys
HVEM bruker tabellen: Feil
BESKRIVELSE av tabellen: Stasjonstabellen for KRO-systemet. Oppdateres automatisk fra klimadatabasen.
BESKRIVELSE av attributtene i tabellen:
nr: Stasjonsnummeret på stasjonen.
navn: Stasjonens navn.
lat: Breddegrad
lon: Lengdegrad
synopnr: Synopnummeret på stasjonen
a:  Bool-verdi for automatstasjon
f:  Bool-verdi for flyplass
h:  Bool-verdi for hybridstasjon
i:   Bool-verdi for piostasjon
m: Bool-verdi for manuell værstasjon
n:  Bool-verdi for nedbørstasjon
t:  Bool-verdi for teststasjon
v:  Bool-verdi for værstasjon                                                              
region: Regiontilhørighet for stasjon (eks. Østlandet, Vestlandet, Nord-Norge)
status: Stasjonsstatus (eks. 3=rød, 2=gul, 1=grønn)


stasjon_inne:
KILDE: Tabellen data i Kvalobs, gjennom overføringsteknologi
HVEM bruker tabellen: Perl-skript bruker denne og obs_pgm for å lage en liste over stasjon_nede
BESKRIVELSE av tabellen: Inneholder stasjonsnr og obstid for stasjoner definert som 'inne'
BESKRIVELSE av attributtene i tabellen:
stationid: Stasjonsnr
obstime: Obstid


stasjon_nede:
KILDE: Stasjon_inne og obs_pgm
HVEM bruker tabellen: Perl-skript bruker denne for å generere web-meldinger til KRO om stasjoner nede. Ikke fullt implementert, gjenstår å
bli enig om kriterier for meldinger.
BESKRIVELSE av tabellen: Inneholder stasjonsnr og obstid for stasjoner definert som 'nede'
BESKRIVELSE av attributtene i tabellen:
stationid: Stasjonsnr
obstime: Obstid

status:
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Stasjon
BESKRIVELSE av tabellen: Inneholder alle statustyper (1,2,3) kombinert med tekstlig forklaring på norsk og engelsk.
BESKRIVELSE av attributtene i tabellen:
nr: statusnummer
farge_norsk_txt: norsk tekst
farge_eng_txt: engelsk tekst


tiltak:
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Feil
BESKRIVELSE av tabellen: Inneholder alle tiltakene som er satt med type, tekst, tidspunkt og signatur og feilsituasjonsnummer
BESKRIVELSE av attributtene i tabellen:
nr: Tiltaksnummeret
tekst: Forklaring på tiltak
tiltaktypenr: Nummeret på tiltakstypen til tiltaket
feilnr: Nummeret på feilsituasjonen som tiltaket tilhører
beskrevet: Bool-verdi for beskrevet/ikke-beskrevet tiltak
beskrivtid: Tidspunkt for beskrevet tiltak
signaturnr: Signaturnummeret til den som har beskrevet tiltak
ferdig: Bool-verdi for rettet/ikke rettet feil
slutttid: Tidspunkt for feil rettet
sluttfarnr: Signaturnummer på den som rettet feilen


tiltakstype:
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Tiltak
BESKRIVELSE av tabellen: Inneholder alle tiltakstypene (eks. 'Kontakt observatør', 'Annet') med tilhørende nummer
BESKRIVELSE av attributtene i tabellen:
nr: nummer på tiltakstypen
tekst: aktuell tiltakstype


 datafeil: Ikke i bruk
kvalobstest: Ikke i bruk
kvalobstype: Ikke i bruk
lokal: Ikke i bruk
parameter: Ikke i bruk

Trykk på 'VELG STASJON' (for meldingsregistrering) på startsiden. Venstre del i nederste høyre ramme (rubrikk Stasjon) i neste vindu.
Vises også:
Trykk på 'Grønne stasjoner' fra startsiden, 'Registrering av ny melding'. Venstre del i nederste høyre ramme.
msynopregside.pl
 Registrering av melding utfra valgt WMO-nr.
Trykk på 'VELG STASJON' (for meldingsregistrering) på startsiden.  Velg stasjon utfra WMO-nr nederst til høyre. Øverste høyre ramme.
Vises også:
Trykk på 'Grønne stasjoner' fra startsiden, 'Registrering av ny melding'. Velg stasjon utfra WMO-nr nederst til høyre. Øverste høyre ramme for meldingsregistrering.

null_tiltak.pl
 Skriptet som håndterer trykk på "Null-tiltak" i Stasjonsstatus-visning (ved rød status
og bekreftet feilsituasjon). "Null-tiltak" blir registrert, feil settes til "rettet" og
stasjonsstatus blir grønn igjen.

 Registrering av ny feilmelding.

regendny_info_ver3.pl Registrering av ny infomelding.

start.pl
 "Startsiden". 'start.pl' er åpningssiden på KRO, her er oversikt over alle stasjonstyper
og deres status (rød, gul, grønn)+infomeldinger. Stasjoner i rubrikken 'VELG STASJON' (der man registrerer feilmelding eller infomelding) er sortert på navn, stigende. Analoge startsider er startb.pl, startc.pl, startd.pl der stasjoner er sortert på navn eller nr. Dette kan velges på startsiden
10 siste feilmeldinger og 10 siste infomeldinger vises.
Link til 'HVORDAN BRUKE KRO' og statistikk på ulike typer feilmeldinger.
Visning av navn etter LDAP-registrering, link til Stasjonsendringer og grønne stasjoner.
startb.pl
 Startside der stasjoner i rubrikken 'VELG STASJON' er sortert på navn, synkende.
startc.pl
 Startside der stasjoner i rubrikken 'VELG STASJON' er sortert på nr, stigende.
startd.pl
 Startside der stasjoner i rubrikken 'VELG STASJON' er sortert på nr, synkende.
startf.pl
 Denne siden viser alle "grønne" stasjoner, stasjoner der det ikke er registrert feilmeldinger på aktuelt tidspunkt.
statistikk.pl
 Viser statistikk for valgt feiltype fra startsiden (Nederst til høyre)
statuside2.pl
 Velg stasjon i en av kategoriene (Automat, Nedbør etc.) på startsiden, trykk på 'Status'. Velg en ny stasjon fra venstre meny i neste skjermbilde. Stasjonsstatusvisning i høyre ramme.
statuside2x.pl Valg av stasjon i 'Mine gule st.'. Valg av annen stasjon i venstre meny i neste skjermbilde. Stasjonsstatus-visning i høyre ramme. 
statuside2y.pl Valg av stasjon i 'Siste 10 feilmeldinger'. Valg av annen stasjon i venstre meny i neste skjermbilde. Resultatet (statuside2y.pl) vises i høyre ramme.
