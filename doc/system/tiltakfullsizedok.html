CREATE TABLE datafeil (

stationid int4 NOT NULL,

obstime timestamptz NOT NULL,

original float8 NOT NULL,

paramid int4 NOT NULL,

tbtime timestamptz NOT NULL,

typeid int4 NOT NULL,

sensor char(1) DEFAULT '0',

level int4,

corrected float8,

controlinfo char(16) DEFAULT '0000000000000000',

useinfo char(16) DEFAULT '0000000000000000',

cfailed text,

metadata text,

tack timestamptz DEFAULT ('now'::text)::timestamp(6) with time zone,

tmesg timestamptz,

CONSTRAINT datafeil_stationid_key UNIQUE (stationid, obstime, paramid, level, sensor, typeid)

);



CREATE TABLE feil (

nr int4 DEFAULT nextval('"feil_nr_seq"'::text) NOT NULL,

stasjonsnr int4 NOT NULL,

feiltypenr int4 NOT NULL,

tiltaksnr int4,

kvittering bool DEFAULT 'f'::bool,

signaturnr int4,

lestid timestamptz,

CONSTRAINT feil_pkey PRIMARY KEY (nr),

CONSTRAINT feil_lestid CHECK (NULL::bool)

);


CREATE TABLE feiltype (

nr int4 DEFAULT nextval('"feiltype_nr_seq"'::text) NOT NULL,

tekst varchar(100) NOT NULL,

CONSTRAINT feiltype_tekst_key UNIQUE (tekst),

CONSTRAINT feiltype_pkey PRIMARY KEY (nr)

);



CREATE TABLE kvalobstest (

nr int4 DEFAULT nextval('"kvalobstest_nr_seq"'::text) NOT NULL,

navn varchar(100) NOT NULL,

antalltekst int4,

tekst varchar(100) DEFAULT 'Ingen beskrivelse foreligger...' NOT NULL,

CONSTRAINT kvalobstest_pkey PRIMARY KEY (nr)

);



CREATE TABLE kvalobstype (

nr int4 NOT NULL,

feiltypenr int4,

CONSTRAINT kvalobstype_pkey PRIMARY KEY (nr)

);



CREATE TABLE lokal (

nr int4 DEFAULT nextval('"lokal_nr_seq"'::text) NOT NULL,

ip_adr inet NOT NULL,

lest_stasjon varchar(100) NOT NULL,

CONSTRAINT lokal_pkey PRIMARY KEY (nr)

);



CREATE TABLE lokal2 (

nr int4 DEFAULT nextval('"lokal2_nr_seq"'::text) NOT NULL,

ip_adr inet NOT NULL,

lest_stasjon varchar(100) NOT NULL,

st_type char(1) NOT NULL,

CONSTRAINT lokal2_nr_key UNIQUE (nr)

);



CREATE TABLE melding (

nr int4 DEFAULT nextval('"melding_nr_seq"'::text) NOT NULL,

tekst varchar(1000) NOT NULL,

feilnr int4 NOT NULL,

tid timestamptz DEFAULT ('now'::text)::timestamp(6) with time zone,

signaturnr int4,

paramnr int4 NOT NULL,

paramverdi numeric(10, 5),

testnr int4 NOT NULL,

testparam numeric[],

CONSTRAINT melding_pkey PRIMARY KEY (nr)

);



CREATE TABLE parameter (

nr int4 NOT NULL,

kode char(10) NOT NULL,

definisjon varchar(100),

kategori varchar(100),

kategoritekst varchar(100),

CONSTRAINT parameter_kode_key UNIQUE (kode),

CONSTRAINT parameter_nr_key UNIQUE (nr),

CONSTRAINT parameter_pkey PRIMARY KEY (nr, kode)

);



CREATE TABLE signatur (

nr int4 DEFAULT nextval('"signatur_nr_seq"'::text) NOT NULL,

navn char(100) NOT NULL,

tilgang int4,

kommentar char(100),

ip_adr inet NOT NULL,

CONSTRAINT signatur_pkey PRIMARY KEY (nr)

);



CREATE TABLE stasjon (

nr int4 NOT NULL,

navn varchar(100) NOT NULL,

lat numeric(6, 4),

lon numeric(6, 4),

synopnr int4,

a bool DEFAULT 'f'::bool,

f bool DEFAULT 'f'::bool,

h bool DEFAULT 'f'::bool,

i bool DEFAULT 'f'::bool,

n bool DEFAULT 'f'::bool,

t bool DEFAULT 'f'::bool,

v bool DEFAULT 'f'::bool,

region varchar(50),

status int4 DEFAULT 1,

CONSTRAINT stasjon_pkey PRIMARY KEY (nr),

CONSTRAINT stasjon_lat CHECK (((lat > '-90.0000'::"numeric") AND (lat < '90.0000'::"numeric"))),

CONSTRAINT stasjon_lon CHECK (((lon > '-90.0000'::"numeric") AND (lon < '90.0000'::"numeric"))),

CONSTRAINT stasjon_region CHECK ((((((((((region = 'Øst'::"varchar") OR NULL::bool) OR (region = 'Vest'::"varchar")) OR (region = 'Nord'::"varchar")) OR (region = 'Sør'::"varchar")) OR (region = 'MidtNorge'::"varchar")) OR (region = 'Maritim'::"varchar")) OR (region = 'Test'::"varchar")) OR (region = 'Arktis'::"varchar"))),

CONSTRAINT "<unnamed>" FOREIGN KEY (status) REFERENCES status (nr) ON DELETE NO ACTION ON UPDATE NO ACTION NOT DEFERRABLE INITIALLY IMMEDIATE

);



CREATE TABLE status (

nr int4 NOT NULL,

farge_norsk_txt varchar(100),

farge_eng_txt varchar(100),

CONSTRAINT status_pkey PRIMARY KEY (nr)

);



CREATE TABLE tiltak (

nr int4 DEFAULT nextval('"tiltak_nr_seq"'::text) NOT NULL,

tekst varchar(1000) NOT NULL,

tiltaktypenr int4 NOT NULL,

feilnr int4 NOT NULL,

beskrevet bool DEFAULT 'f'::bool,

beskrivtid timestamptz,

signaturnr int4 NOT NULL,

ferdig bool DEFAULT 'f'::bool,

slutttid timestamptz,

sluttfarnr int4,

CONSTRAINT tiltak_pkey PRIMARY KEY (nr),

CONSTRAINT "<unnamed>" FOREIGN KEY (tiltaktypenr) REFERENCES tiltakstype (nr) ON DELETE RESTRICT ON UPDATE NO ACTION NOT DEFERRABLE INITIALLY IMMEDIATE,

CONSTRAINT "<unnamed>" FOREIGN KEY (signaturnr) REFERENCES signatur (nr) ON DELETE RESTRICT ON UPDATE NO ACTION NOT DEFERRABLE INITIALLY IMMEDIATE

);



CREATE TABLE tiltakstype (

nr int4 DEFAULT nextval('"tiltakstype_nr_seq"'::text) NOT NULL,

tekst varchar(100) NOT NULL,

CONSTRAINT tiltakstype_pkey PRIMARY KEY (nr)

);





DOKUMENTASJON AV TABELLENE.

For tabeller i bruk pr. idag:
Vi har de statiske tabellene feiltype, tiltakstype, status og signatur. Signatur vil dog endres når nye personer logger seg på.
De dynamiske tabellene er feil (inneholder alle feilsituasjoner), melding (alle meldingstekster generert), tiltak (alle beskrevne tiltak, boolsk verdi for bekreftet feil/feil rettet), stasjon (inneholder alle stasjoner, oppdateres mot st_info i klimadatabasen) og lokal2 (benyttes som midlertidig lagring av valgt stasjonsnavn til bruk i skript).

datafeil: (Ikke i bruk enda, i bruk når overføringer av feilposter fra Kvalobs starter)
KILDE: Kvalobs
HVEM bruker tabellen: Perlskript for å generere tekstlig feilmelding til visning i KRO 
BESKRIVELSE av tabellen: Dette er en kopi av tabellen 'data' i Kvalobs som inneholder èn post for hver inkommende observasjon.
I vårt tilfelle blir det èn post for hver observasjon som er flagget med indikasjoner av mistenksomhet/feil
BESKRIVELSE av attributtene i tabellen:
stationid: Stasjonsnummer
obstime: Observasjonstid. Tidspunkt for observasjonens gyldighet.
original: Observert parameterverdi
paramid: Kvalobs-internt parameternummer
tbtime: Tidspunkt for når posten ble generert i Kvalobsdb.
typeid: Betegnelse på meldingsformat. Samme observasjon kan foreligge i flere
ulike formater.
sensor: Sensornummer der samme parameter observeres ved flere sensorer.
level: Nivå der samme parameter observeres i ulik høyde eller dybde,
corrected: Observert verdi evt. endret i kontrollene.
controlinfo: Kodet informasjon om resultatet av kontrollene, koden er 64 bit.
useinfo: Kodet kvalitetsinformasjon om observasjonen, koden er 64 bit.
cfailed: tester som har gitt utslag; syntaksen for denne kolonnen er
qcx: language der en får verdiene i fra tabellen checks.
tack: Tidspunktet for når posten ble generert i Tiltaksdatabasen.
tmesg:


feil:
 KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Alle de andre tabellene i tiltaksdatabasen
BESKRIVELSE av tabellen: Dette er den sentrale tabellen i tiltaksdatabasen , som kobler feiltype, stasjonsnummer, melding, tiltak og signatur.
BESKRIVELSE av attributtene i tabellen:
nr: Nummeret på feilsituasjonen
stasjonsnr: Stasjonsnummeret
feiltypenr: Nummeret på feiltypen
tiltaksnr: Nummeret på eventuelt tiltak som er beskrevet
kvittering: Boolsk verdi ('t'/'f') for om feilsituasjonen er bekreftet eller ikke
signaturnr: Signaturnummer på vedkommende som har bekreftet feilsituasjonen
lestid: Tidspunkt for når feilsituasjonen er bekreftet (er tom hvis feilsituasjonen ikke er bekreftet)


feiltype:
 KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Feil
BESKRIVELSE av tabellen: Inneholder alle feiltypekategoriene med tilhørende nr.
BESKRIVELSE av attributtene i tabellen:
nr: nummer på feiltype
tekst: aktuell feiltype


kvalobstest: Pr. idag ikke i bruk
kvalobstype: Pr. idag ikke i bruk
lokal: Pr. idag ikke i bruk


lokal2:
 KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Perlskriptet som viser resultatet av valget ditt
BESKRIVELSE av tabellen: Brukes som mellomtabell for valgt stasjon,
dersom man trykker på "Registrering av ny melding", "Historikk" etc. så "huskes" stasjonen
man har valgt ved hjelp av denne tabellen (stasjonsvalget blir altså lagret midlertidig)
BESKRIVELSE av attributtene i tabellen:
nr: Løpenummer
ip_adr: IP-addressen til vedkommende som er inne på aktuell stasjon
lest_stasjon: Aktuell stasjon man har valgt
st_type: Type stasjon


melding:
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Perlskript for å vise registrert melding
BESKRIVELSE av tabellen: Denne tabellen inneholder alle meldinger som er registrert, med signatur, feilnummer og tidspunkt
BESKRIVELSE av attributtene i tabellen:
nr: løpenummeret på meldingen
tekst: meldingsteksten
feilnr: nummeret på feilsituasjonen som meldingen tilhører
tid: tid for innkommet melding
signaturnr: signaturnummeret til meldingskilden
paramnr:          Ikke i bruk
paramverdi:     Ikke i bruk
testnr:               Ikke i bruki 
testparam:        Ikke i bruk


parameter: (pr. 15/10-2003 ikke i bruk, men benyttes så snart overføring av feil fra Kvalobs starter)
KILDE: Kopiert fra 'Param'-tabellen i Kvalobs
HVEM bruker tabellen: Meldingstabellen
BESKRIVELSE av tabellen: Tabellen inneholder alle parametre i Kvalobs og tilhørende parameternummer, kode og kategori.
Denne benyttes så snart overføring av feil fra Kvalobs starter, og benyttes som oppslag til parametertekst for generert feilmelding. 
BESKRIVELSE av attributtene i tabellen:
nr: Parameternummer
kode: Parameterkode
definisjon: Parameterdefinisjon
kategori: Parameterkategori
kategoritekst: Kategoritekst på parameteren


signatur: (Denne tabellen skal modifiseres med LDAP-tilknytning, er under utarbeidelse nå. Det som står nå er altså slik situasjonen fungerer pr. idag)
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Feil, Melding, Tiltak
BESKRIVELSE av tabellen: Signaturtabell. Inneholder alle registrerte brukere med selvvalgt brukernavn, tilhørende ip-adresse og nummer
BESKRIVELSE av attributtene i tabellen:
nr: Signaturnummer
navn: Brukernavn
tilgang: Tall som angir rettigheter (1,2,3,4,5)
kommentar: Kommentarfelt
ip_adr: Brukerens ip-addresse


stasjon:
KILDE: Klimadatabasen, senere St_Info_Sys
HVEM bruker tabellen: Feil
BESKRIVELSE av tabellen: Stasjonstabellen for KRO-systemet. Oppdateres automatisk fra klimadatabasen.
BESKRIVELSE av attributtene i tabellen:
nr: Stasjonsnummeret på stasjonen.
navn: Stasjonens navn.
lat: Breddegrad
lon: Lengdegrad
synopnr: Synopnummeret på stasjonen
a:  Bool-verdi for automatstasjon
f:  Bool-verdi for flyplass
h:  Bool-verdi for hybridstasjon
i:   Bool-verdi for piostasjon
m: Bool-verdi for manuell værstasjon
n:  Bool-verdi for nedbørstasjon
t:  Bool-verdi for teststasjon
v:  Bool-verdi for værstasjon                                                              
region: Regiontilhørighet for stasjon (eks. Østlandet, Vestlandet, Nord-Norge)
status: Stasjonsstatus (eks. 3=rød, 2=gul, 1=grønn)


status:
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Stasjon
BESKRIVELSE av tabellen: Inneholder alle statustyper (1,2,3) kombinert med tekstlig forklaring på norsk og engelsk.
BESKRIVELSE av attributtene i tabellen:
nr: statusnummer
farge_norsk_txt: norsk tekst
farge_eng_txt: engelsk tekst


tiltak:
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Feil
BESKRIVELSE av tabellen: Inneholder alle tiltakene som er satt med type, tekst, tidspunkt og signatur og feilsituasjonsnummer
BESKRIVELSE av attributtene i tabellen:
nr: Tiltaksnummeret
tekst: Forklaring på tiltak
tiltaktypenr: Nummeret på tiltakstypen til tiltaket
feilnr: Nummeret på feilsituasjonen som tiltaket tilhører
beskrevet: Bool-verdi for beskrevet/ikke-beskrevet tiltak
beskrivtid: Tidspunkt for beskrevet tiltak
signaturnr: Signaturnummeret til den som har beskrevet tiltak
ferdig: Bool-verdi for rettet/ikke rettet feil
slutttid: Tidspunkt for feil rettet
sluttfarnr: Signaturnummer på den som rettet feilen


tiltakstype:
KILDE: Tiltaksdatabasen
HVEM bruker tabellen: Tiltak
BESKRIVELSE av tabellen: Inneholder alle tiltakstypene (eks. 'Kontakt observatør', 'Annet') med tilhørende nummer
BESKRIVELSE av attributtene i tabellen:
nr: nummer på tiltakstypen
tekst: aktuell tiltakstype

Forklaring av hvert perl-skript/hver webside:

kart2.pl "kartsiden". 'kart2.pl' er åpningssiden på KRO, her er oversikt over alle stasjonstyper
og deres status (rød, gul, grønn) Man har knapp for å registrere brukernavn og melding.
Bn.html
 Dette er siden for å registrere ny bruker/endre brukernavn
(ved trykk på "Hvis du ikke er........" fra kartsiden
frames.html
 Rammesett for å velge stasjon ut fra region eller fra komplett stasjonsliste.
(Ved trykk på "Velg region og stasjon" fra kartsiden). Valg av region og stasjon øverst,
meny til venstre og stasjonsstatusvisning i nederste høyre ramme.
histframes.html Rammesett for å vise historikk, ut fra trykk på knappen "Vis historikk" i Stasjonsstatus-
visning. Meny på venstre side, historikk-visning i høyre ramme.
index.html
 Siden som "redirigerer" addressen fra http://modon/tiltak til
http://modon.oslo.dnmi.no/cgi-bin/tiltak/kart2.pl  (Dette for å ha en lettere startaddresse)
meldingframes.html
 Rammesett for å registrere ny melding på selvvalgt stasjon (Ved trykk på "Registrering
av ny melding" fra kartsiden). Valg av region, stasjon øverst, meny til venstre og
meldingsregistrering i nederste høyre ramme.
meldingframes1.html
 For å registrere melding på forhåndsvalgt stasjon fra kartsiden (Ved trykk på
"Registrering av ny melding" i neste vindu). Meny til venstre, meldingsregistrering i
høyre ramme.
met.gif
 Bilde-logo for Meteorologisk institutt.

kartsiden (røde). Meny på venstre side, stasjonsstatusvisning i høyre ramme.
